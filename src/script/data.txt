** 외부 파일을 전송하는 법
----------------------------------------------------------------
전송 보낼 파일
const apple = 10;
const apple2 = 10;

페이지 하단에 export 작성
export default apple

두가지 이상의 데이터 전송
export {apple, apple2}
----------------------------------------------------------------
받을 파일
import 데이터를 받는 변수 from './파일경로';

두가지 이상의 데이터 받기
중괄호로 데이터 전송했으면 중괄호로 받아야함!! 
import {변수1, 변수2} from  './파일경로';
